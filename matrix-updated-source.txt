    function buildResourcesSection(p, assetFilter) {
        var html = '';
        
        // Show workspaces with IDs (only if filter is 'all' or 'Workspace')
        if (p.workspaces && p.workspaces.length > 0 && (assetFilter === 'all' || assetFilter === 'Workspace')) {
            html += '<div class="mv-resources-section"><div class="mv-resource-group">';
            html += '<div class="mv-resource-label">📁 Workspaces:</div>';
            for (var i = 0; i < p.workspaces.length; i++) {
                html += '<div class="mv-resource-item">';
                html += '<span class="mv-resource-name">' + p.workspaces[i] + '</span>';
                if (p.workspace_ids && p.workspace_ids[i]) {
                    html += '<span class="mv-resource-id">' + p.workspace_ids[i] + '</span>';
                }
                html += '</div>';
            }
            html += '</div>';
            
            // Show connected data collections for workspaces
            if (p.connected_resources && Object.keys(p.connected_resources).length > 0) {
                html += '<div class="mv-connections">';
                html += '<div class="mv-connections-label">🔗 Connected Data Collections:</div>';
                for (var wsId in p.connected_resources) {
                    var collections = p.connected_resources[wsId];
                    if (collections && collections.length > 0) {
                        html += '<div class="mv-connection-item">' + wsId + ' → ' + collections.join(', ') + '</div>';
                    }
                }
                html += '</div>';
            }
            html += '</div>';
        }
        
        // Show data collections with IDs (only if filter is 'all' or 'Data Collection')
        if (p.data_collections && p.data_collections.length > 0 && (assetFilter === 'all' || assetFilter === 'Data Collection')) {
            html += '<div class="mv-resources-section"><div class="mv-resource-group">';
            html += '<div class="mv-resource-label">📦 Data Collections:</div>';
            for (var i = 0; i < p.data_collections.length; i++) {
                html += '<div class="mv-resource-item">';
                html += '<span class="mv-resource-name">' + p.data_collections[i] + '</span>';
                if (p.data_collection_ids && p.data_collection_ids[i]) {
                    html += '<span class="mv-resource-id">' + p.data_collection_ids[i] + '</span>';
                }
                html += '</div>';
            }
            html += '</div>';
            
            // Show connected workspaces for data collections
            if (p.connected_resources && Object.keys(p.connected_resources).length > 0) {
                var connectedToThis = [];
                for (var wsId in p.connected_resources) {
                    connectedToThis.push(wsId);
                }
                if (connectedToThis.length > 0) {
                    html += '<div class="mv-connections">';
                    html += '<div class="mv-connections-label">🔗 Connected to Workspaces:</div>';
                    html += '<div class="mv-connection-item">' + connectedToThis.join(', ') + '</div>';
                    html += '</div>';
                }
            }
            html += '</div>';
        }
        
        return html;
    }

    function renderTable() {
        var tbody = document.getElementById('mvTableBody');
        var filtered = projects.filter(function(p) {
            var matchesAsset = currentFilters.assetType === 'all' || p.assetType === currentFilters.assetType;
            var matchesAccess = currentFilters.accessLevel === 'all' || p.accessLevel === currentFilters.accessLevel;
            var searchTerm = currentFilters.search.toLowerCase();
            var matchesSearch = !searchTerm ||
                p.name.toLowerCase().indexOf(searchTerm) !== -1 ||
                p.desc.toLowerCase().indexOf(searchTerm) !== -1 ||
                p.owner.toLowerCase().indexOf(searchTerm) !== -1;
            return matchesAsset && matchesAccess && matchesSearch;
        });

        if (currentSort.column) {
            filtered.sort(function(a, b) {
                var aVal = a[currentSort.column];
                var bVal = b[currentSort.column];

                // Special sorting logic for owner column
                if (currentSort.column === 'owner') {
                    // Function to determine sort priority
                    function getOwnerPriority(owner) {
                        if (owner === 'TBD') return 3; // TBD goes last
                        if (owner.indexOf('[Squad]') === 0) return 2; // Squad names go second
                        return 1; // Individual person names go first
                    }

                    var aPriority = getOwnerPriority(aVal);
                    var bPriority = getOwnerPriority(bVal);

                    // If different priorities, sort by priority
                    if (aPriority !== bPriority) {
                        return currentSort.direction === 'asc' ?
                            aPriority - bPriority :
                            bPriority - aPriority;
                    }

                    // Same priority - sort alphabetically by first name
                    var aFirstName = aVal.split(',')[0].trim().replace('@', '').replace('[Squad] ', '');
                    var bFirstName = bVal.split(',')[0].trim().replace('@', '').replace('[Squad] ', '');
                    var comparison = aFirstName.localeCompare(bFirstName);
                    return currentSort.direction === 'asc' ? comparison : -comparison;
                }

                var comparison = aVal.localeCompare(bVal);
                return currentSort.direction === 'asc' ? comparison : -comparison;
            });
        }

        tbody.innerHTML = filtered.map(function(p) {
            return '<tr>' +
                '<td><div class="mv-project-name"><a href="' + p.link + '" class="mv-project-link" target="_blank" rel="noopener noreferrer">' + p.name + '</a></div><p class="mv-project-desc">' + p.desc + '</p>' + buildResourcesSection(p, currentFilters.assetType) + '</td>' +
                '<td><span class="mv-badge mv-badge-asset">' + p.assetType + '</span></td>' +
                '<td><span class="mv-badge mv-badge-access">' + p.accessLevel + '</span></td>' +
                '</tr>';
        }).join('');

        // Update stats
        document.getElementById('mvVisibleCount').textContent = filtered.length;
        var workspaceCount = filtered.filter(function(p) { return p.assetType === 'Workspace'; }).length;
        var collectionCount = filtered.filter(function(p) { return p.assetType === 'Data Collection'; }).length;
        document.getElementById('mvWorkspaceCount').textContent = workspaceCount;
        document.getElementById('mvCollectionCount').textContent = collectionCount;
    }

    function mvSortTable(column) {
        if (currentSort.column === column) {
            currentSort.direction = currentSort.direction === 'asc' ? 'desc' : 'asc';
        } else {
            currentSort.column = column;
            currentSort.direction = 'asc';
        }

        document.querySelectorAll('.mv-table th.sortable').forEach(function(h) {
            h.classList.remove('sorted-asc', 'sorted-desc');
        });
        var th = document.querySelector('.mv-table th[data-sort="' + column + '"]');
        th.classList.add(currentSort.direction === 'asc' ? 'sorted-asc' : 'sorted-desc');

        renderTable();
    }

    function mvFilterAsset(pill, filter) {
        document.querySelectorAll('#mvAssetFilters .mv-pill').forEach(function(p) {
            p.classList.remove('active');
        });
        pill.classList.add('active');
        currentFilters.assetType = filter;
        renderTable();
    }


    function mvFilterAccess(pill, filter) {
        document.querySelectorAll('#mvAccessFilters .mv-pill').forEach(function(p) {
            p.classList.remove('active');
        });
        pill.classList.add('active');
        currentFilters.accessLevel = filter;
        renderTable();
    }

    function mvToggleView(view) {
        var table = document.getElementById('mvProjectTable');
        var buttons = document.querySelectorAll('.mv-view-btn');
        buttons.forEach(function(btn) {
            btn.classList.remove('active');
        });
        event.target.classList.add('active');

        if (view === 'compact') {
            table.classList.add('mv-compact');
        } else {
            table.classList.remove('mv-compact');
        }
    }

    window.mvSortTable = mvSortTable;
    window.mvFilterAsset = mvFilterAsset;
    window.mvFilterAccess = mvFilterAccess;
    window.mvToggleView = mvToggleView;

    document.getElementById('mvSearchInput').addEventListener('input', function(e) {
        currentFilters.search = e.target.value;
        renderTable();
    });

    renderTable();
})();
</script>
